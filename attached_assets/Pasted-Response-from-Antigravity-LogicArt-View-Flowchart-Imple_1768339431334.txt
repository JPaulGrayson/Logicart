2. The exact JSX code for the button:
tsx
{/* View Flowchart Button - LogicArt visualization */}
<button
  onClick={async () => {
    // Find the main app module script (Vite uses .tsx/.ts, bundled apps use .js)
    const scriptTag = document.querySelector('script[type="module"][src*="/src/"]:not([src*="logic.art"]), script[src*=".js"]:not([src*="logic.art"])') as HTMLScriptElement | null;
    if (!scriptTag) { 
      alert('No script found'); 
      return; 
    }
    
    try {
      const response = await fetch(scriptTag.src);
      const code = await response.text();
      
      if ((window as any).LogicArt && (window as any).LogicArt.visualize) {
        (window as any).LogicArt.visualize(code, 'AppCode');
      } else {
        alert('LogicArt not loaded');
      }
    } catch (err: any) {
      alert('Failed to load code: ' + err.message);
    }
  }}
  className="flex items-center gap-1 flex-shrink-0 px-3 py-2 sm:px-2 sm:py-1.5 rounded-lg text-xs font-medium bg-purple-600 text-white hover:bg-purple-700 transition-colors"
  data-testid="button-view-flowchart"
  aria-label="View Flowchart"
>
  <GitBranch className="w-4 h-4" />
  <span className="hidden sm:inline">Flowchart</span>
</button>
