Impact: Variable declaration labels may show as const undefined = ... instead of const [count, setCount] = ... for destructured variables.

// Before (bridge/src/parser.ts line 621):
const label = `${stmt.kind} ${decl.id.name} = ...`;
// After (from docs/bridge/src/parser.ts lines 747-766):
let varName: string;
if (decl.id.type === 'Identifier') {
  varName = decl.id.name;
} else if (decl.id.type === 'ArrayPattern') {
  const names = decl.id.elements
    ?.map((el: any) => el?.name || '_')
    .join(', ') || '...';
  varName = `[${names}]`;
} else if (decl.id.type === 'ObjectPattern') {
  const names = decl.id.properties
    ?.map((prop: any) => prop.key?.name || '_')
    .join(', ') || '...';
  varName = `{ ${names} }`;
} else {
  varName = 'variable';
}
const label = `${stmt.kind} ${varName} = ...`;
